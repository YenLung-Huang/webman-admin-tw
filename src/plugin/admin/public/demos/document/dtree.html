開發環境
Dtree 提供 樹狀態 的資料結構展示

顯示代碼

								<link href="component/pear/css/pear.css" rel="stylesheet" />
								 并
								<script src="component/layui/layui.js"></script>
								 并
								<script src="component/pear/pear.js"></script>
嵌套資料格式

    顯示代碼

    								layui.use(['dtree'], function () {
								    dtree = layui.dtree;
								 
								    dtree.render({
								        elem: "#demoTree",
								        initLevel: "1",
								        method: 'get',
								        url: "admin/data/dtree.json"
								    });
								});
    嵌套資料格式

      顯示代碼

      								layui.use(['dtree'], function () {
								    dtree = layui.dtree;
								 
								    dtree.render({
								        elem: "#demoTree",
								        initLevel: "1",
								        method: 'get',
								        url: "admin/data/dtree.json",
								        dataFormat: "list", 
								        checkbar: true
								    });
								});
